프로그래머스 PCCP 모의고사 1회 


```java
class Solution {
    static int max = 0;

    public int solution(int[][] ability) {
        boolean[] visitedMan = new boolean[ability.length];
        boolean[] visitedSports = new boolean[ability[0].length];

        dfs(ability, 0, 0, 0, visitedMan, visitedSports);

        return max;
    }

    private static void dfs(int[][] ability, int L, int start, int sum, boolean[] visitedMan, boolean[] visitedSports) {

        if (L == ability[0].length) {
            max = Math.max(max, sum);
        } else {
            for (int i = start; i < ability[0].length; i++) {
                if (!visitedSports[i]) {
                    visitedSports[i] = true;
                    for (int j = 0; j < ability.length; j++) {
                        if (!visitedMan[j]) {
                            visitedMan[j] = true;
                            dfs(ability, L + 1, i+1,sum + ability[j][i], visitedMan, visitedSports);
                            visitedMan[j] = false;
                        }
                    }
                    visitedSports[i] = false;
                }
            }
        }
    }
}
```
